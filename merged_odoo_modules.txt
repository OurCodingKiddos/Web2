                }
            });
            
            // Draw coins
            coinsArray.forEach(coin => {
                if (!coin.collected) {
                    const bounce = Math.sin(coin.animation) * 5;
                    ctx.fillStyle = '#FFD700';
                    ctx.beginPath();
                    ctx.arc(
                        coin.x - cameraX + coin.width/2, 
                        coin.y + bounce + coin.height/2, 
                        coin.width/2, 0, Math.PI * 2
                    );
                    ctx.fill();
                    
                    // Coin shine
                    ctx.fillStyle = '#FFF8DC';
                    ctx.beginPath();
                    ctx.arc(
                        coin.x - cameraX + coin.width/2 - 3, 
                        coin.y + bounce + coin.height/2 - 3, 
                        coin.width/6, 0, Math.PI * 2
                    );
                    ctx.fill();
                }
            });
            
            // Draw enemies
            enemies.forEach(enemy => {
                ctx.fillStyle = enemy.color;
                ctx.fillRect(enemy.x - cameraX, enemy.y, enemy.width, enemy.height);
                
                // Enemy shell
                ctx.fillStyle = '#4B0082';
                ctx.fillRect(enemy.x - cameraX + 5, enemy.y + 5, enemy.width - 10, enemy.height - 10);
                
                // Enemy eyes
                ctx.fillStyle = 'white';
                ctx.fillRect(enemy.x - cameraX + 8, enemy.y + 10, 6, 6);
                ctx.fillRect(enemy.x - cameraX + 18, enemy.y + 10, 6, 6);
                
                ctx.fillStyle = 'black';
                ctx.fillRect(enemy.x - cameraX + 10, enemy.y + 12, 2, 2);
                ctx.fillRect(enemy.x - cameraX + 20, enemy.y + 12, 2, 2);
            });
            
            // Draw flagpole
            if (flagpole) {
                ctx.fillStyle = flagpole.color;
                ctx.fillRect(flagpole.x - cameraX, flagpole.y, flagpole.width, flagpole.height);
                
                // Flag
                ctx.fillStyle = '#FFFFFF';
                ctx.beginPath();
                ctx.moveTo(flagpole.x - cameraX + flagpole.width, flagpole.y);
                ctx.lineTo(flagpole.x - cameraX + flagpole.width + 40, flagpole.y + 20);
                ctx.lineTo(flagpole.x - cameraX + flagpole.width, flagpole.y + 40);
                ctx.closePath();
                ctx.fill();
            }
            
            // Draw player
            drawPlayer();
        }
        
        // Draw background elements
        function drawBackground() {
            // Mountains
            ctx.fillStyle = '#5D4037';
            ctx.beginPath();
            ctx.moveTo(0 - cameraX * 0.5, 450);
            ctx.lineTo(200 - cameraX * 0.5, 250);
            ctx.lineTo(400 - cameraX * 0.5, 450);
            ctx.fill();
            
            ctx.beginPath();
            ctx.moveTo(300 - cameraX * 0.5, 450);
            ctx.lineTo(500 - cameraX * 0.5, 200);
            ctx.lineTo(700 - cameraX * 0.5, 450);
            ctx.fill();
            
            // Clouds
            ctx.fillStyle = 'white';
            ctx.beginPath();
            ctx.arc(100 - cameraX * 0.3, 80, 30, 0, Math.PI * 2);
            ctx.arc(130 - cameraX * 0.3, 70, 40, 0, Math.PI * 2);
            ctx.arc(160 - cameraX * 0.3, 80, 30, 0, Math.PI * 2);
            ctx.fill();
            
            ctx.beginPath();
            ctx.arc(500 - cameraX * 0.3, 120, 30, 0, Math.PI * 2);
            ctx.arc(530 - cameraX * 0.3, 110, 40, 0, Math.PI * 2);
            ctx.arc(560 - cameraX * 0.3, 120, 30, 0, Math.PI * 2);
            ctx.fill();
        }
        
        // Draw player with details
        function drawPlayer() {
            // Player body
            ctx.fillStyle = player.color;
            ctx.fillRect(player.x - cameraX, player.y, player.width, player.height);
            
            // Player overalls
            ctx.fillStyle = '#1E40AF';
            ctx.fillRect(player.x - cameraX + 5, player.y + 25, player.width - 10, player.height - 25);
            
            // Player hat
            ctx.fillStyle = player.color;
            ctx.fillRect(player.x - cameraX - 5, player.y, player.width + 10, 10);
            ctx.fillRect(player.x - cameraX + player.width/2 - 5, player.y - 10, 10, 10);
            
            // Player face
            ctx.fillStyle = '#FBC02D';
            ctx.fillRect(player.x - cameraX + 5, player.y + 10, player.width - 10, 15);
            
            // Player eyes
            ctx.fillStyle = 'white';
            if (player.facingRight) {
                ctx.fillRect(player.x - cameraX + 18, player.y + 12, 6, 6);
            } else {
                ctx.fillRect(player.x - cameraX + 8, player.y + 12, 6, 6);
            }
            
            ctx.fillStyle = 'black';
            if (player.facingRight) {
                ctx.fillRect(player.x - cameraX + 20, player.y + 14, 2, 2);
            } else {
                ctx.fillRect(player.x - cameraX + 10, player.y + 14, 2, 2);
            }
            
            // Player mustache
            ctx.fillStyle = '#000000';
            ctx.fillRect(player.x - cameraX + 8, player.y + 22, player.width - 16, 3);
        }
        
        // Main game loop
        function gameLoop() {
            if (!gameRunning) return;
            
            update();
            render();
            
            requestAnimationFrame(gameLoop);
        }
        
        // Initial render
        initGameObjects();
        render();
    </script>
